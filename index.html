import random print("🎯 Welcome to the Guessing Game!") print("I have picked a number between 1 and 10.") print("Can you guess it? 🤔") secret_number = random.randint(1, 10) tries = 0 while True: user_guess = input("Enter your guess (1-10): ") if not user_guess.isdigit(): print("⚠ Please enter a valid number!") continue user_guess = int(user_guess) tries += 1 if user_guess == secret_number: print(f"✅ Correct! You guessed it in {tries} tries 🎉") break elif user_guess < secret_number: print("📉 Too low! Try again.") else: print("📈 Too high! Try again.")
